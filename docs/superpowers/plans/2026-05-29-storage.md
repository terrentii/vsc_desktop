# Локальное хранилище (mys_storage) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Построить слой локального зашифрованного хранилища `mys_storage` на SQLCipher: вывод ключа БД из мастер-пароля (Argon2id), полная схема, единый фасад `Storage` с CRUD и rekey.

**Architecture:** Одна БД, зашифрованная целиком через SQLCipher (AES-256). Ключ выводится из мастер-пароля через Argon2id; параметры вывода лежат в несекретном сайдкар-файле рядом с БД. Фасад `Storage` скрывает SQL и предоставляет методы для настроек, контактов, идентичностей, диалогов, сообщений и состояния Double Ratchet. Сериализация `RatchetState` живёт в `mys_crypto.ratchet`.

**Tech Stack:** Python 3.13, `sqlcipher3-binary`, `argon2-cffi` (через `mys_crypto.primitives.argon2id`), `pytest`.

def _b64(b: bytes | None) -> str | None:
    return base64.b64encode(b).decode("ascii") if b is not None else None


def _unb64(s: str | None) -> bytes | None:
    return base64.b64decode(s) if s is not None else None


def serialize_state(state: RatchetState) -> bytes:
    data = {
        "dhs": [_b64(state.dhs[0]), _b64(state.dhs[1])],
        "dhr": _b64(state.dhr),
        "rk": _b64(state.rk),
        "cks": _b64(state.cks),
        "ckr": _b64(state.ckr),
        "ns": state.ns,
        "nr": state.nr,
        "pn": state.pn,
        "mkskipped": [
            [_b64(dh), n, _b64(mk)] for (dh, n), mk in state.mkskipped.items()
        ],
    }
    return json.dumps(data).encode("utf-8")


def deserialize_state(blob: bytes) -> RatchetState:
    data = json.loads(blob.decode("utf-8"))
    return RatchetState(
        dhs=(_unb64(data["dhs"][0]), _unb64(data["dhs"][1])),
        dhr=_unb64(data["dhr"]),
        rk=_unb64(data["rk"]),
        cks=_unb64(data["cks"]),
        ckr=_unb64(data["ckr"]),
        ns=data["ns"],
        nr=data["nr"],
        pn=data["pn"],
        mkskipped={
            (_unb64(dh), n): _unb64(mk) for dh, n, mk in data["mkskipped"]
        },
    )
```

import base64
import json
import os
from dataclasses import dataclass

from mys_crypto.primitives import argon2id

DEFAULT_ARGON2 = {"time": 3, "memory": 65536, "parallelism": 4}
KEY_LEN = 32
SALT_LEN = 16
PARAMS_VERSION = 1


@dataclass
class KeyParams:
    version: int
    salt: bytes
    argon2: dict
    key_len: int


def sidecar_path(db_path: str) -> str:
    return db_path + ".keyparams"


def generate_key_params() -> KeyParams:
    return KeyParams(
        version=PARAMS_VERSION,
        salt=os.urandom(SALT_LEN),
        argon2=dict(DEFAULT_ARGON2),
        key_len=KEY_LEN,
    )


def write_key_params(db_path: str, params: KeyParams) -> None:
    data = {
        "version": params.version,
        "salt": base64.b64encode(params.salt).decode("ascii"),
        "argon2": params.argon2,
        "key_len": params.key_len,
    }
    with open(sidecar_path(db_path), "w", encoding="utf-8") as f:
        json.dump(data, f)


def read_key_params(db_path: str) -> KeyParams:
    with open(sidecar_path(db_path), encoding="utf-8") as f:
        data = json.load(f)
    return KeyParams(
        version=data["version"],
        salt=base64.b64decode(data["salt"]),
        argon2=data["argon2"],
        key_len=data["key_len"],
    )


def derive_db_key(password: bytes, params: KeyParams) -> bytes:
    a = params.argon2
    return argon2id(
        password,
        params.salt,
        params.key_len,
        time_cost=a["time"],
        memory_cost=a["memory"],
        parallelism=a["parallelism"],
    )
```

SCHEMA_VERSION = 1

DDL = """
CREATE TABLE settings (
  key   TEXT PRIMARY KEY,
  value TEXT NOT NULL
);

CREATE TABLE identities (
  id          INTEGER PRIMARY KEY,
  kind        TEXT NOT NULL,
  public_key  BLOB,
  private_key BLOB,
  meta        TEXT
);

CREATE TABLE contacts (
  id          INTEGER PRIMARY KEY,
  public_key  BLOB NOT NULL UNIQUE,
  fingerprint TEXT NOT NULL,
  alias       TEXT,
  created_at  INTEGER NOT NULL
);

CREATE TABLE conversations (
  id         INTEGER PRIMARY KEY,
  mode       TEXT NOT NULL,
  peer_key   BLOB,
  room       TEXT,
  created_at INTEGER NOT NULL
);

CREATE TABLE messages (
  id              INTEGER PRIMARY KEY,
  conversation_id INTEGER NOT NULL REFERENCES conversations(id),
  direction       TEXT NOT NULL,
  body            BLOB NOT NULL,
  status          TEXT NOT NULL,
  created_at      INTEGER NOT NULL
);

CREATE INDEX idx_messages_conv ON messages(conversation_id, id);

CREATE TABLE ratchet_state (
  conversation_id INTEGER PRIMARY KEY REFERENCES conversations(id),
  state           BLOB NOT NULL
);
"""


def apply_schema(conn) -> None:
    conn.executescript(DDL)
    conn.execute(
        "INSERT INTO settings(key, value) VALUES('schema_version', ?)",
        (str(SCHEMA_VERSION),),
    )
    conn.commit()


def get_schema_version(conn) -> int:
    row = conn.execute(
        "SELECT value FROM settings WHERE key='schema_version'"
    ).fetchone()
    return int(row[0]) if row else 0


def migrate(conn) -> None:
    version = get_schema_version(conn)
    if version > SCHEMA_VERSION:
        raise RuntimeError(
            f"DB schema version {version} is newer than supported {SCHEMA_VERSION}"
        )
    # Будущие миграции применяются здесь по порядку, по одной на версию.
```

from sqlcipher3 import dbapi2 as sqlcipher

from . import keyfile, schema


class WrongPassword(Exception):
    """Поднимается при попытке открыть БД с неверным мастер-паролем."""


def _apply_key(conn, key: bytes) -> None:
    conn.execute(f"PRAGMA key = \"x'{key.hex()}'\"")


class Storage:
    def __init__(self, conn, path: str):
        self._conn = conn
        self._path = path
        self._conn.row_factory = sqlcipher.Row

    @classmethod
    def create(cls, path: str, password: str) -> "Storage":
        params = keyfile.generate_key_params()
        keyfile.write_key_params(path, params)
        key = keyfile.derive_db_key(password.encode("utf-8"), params)
        conn = sqlcipher.connect(path)
        _apply_key(conn, key)
        schema.apply_schema(conn)
        return cls(conn, path)

    @classmethod
    def open(cls, path: str, password: str) -> "Storage":
        params = keyfile.read_key_params(path)
        key = keyfile.derive_db_key(password.encode("utf-8"), params)
        conn = sqlcipher.connect(path)
        _apply_key(conn, key)
        try:
            conn.execute("SELECT count(*) FROM sqlite_master").fetchone()
        except sqlcipher.DatabaseError as exc:
            conn.close()
            raise WrongPassword() from exc
        schema.migrate(conn)
        return cls(conn, path)

    def close(self) -> None:
        self._conn.close()

    def get_setting(self, key: str):
        row = self._conn.execute(
            "SELECT value FROM settings WHERE key=?", (key,)
        ).fetchone()
        return row["value"] if row else None

    def set_setting(self, key: str, value: str) -> None:
        self._conn.execute(
            "INSERT INTO settings(key, value) VALUES(?, ?) "
            "ON CONFLICT(key) DO UPDATE SET value=excluded.value",
            (key, value),
        )
        self._conn.commit()
```

    def create_conversation(self, mode: str, peer_key: bytes | None = None, room: str | None = None) -> int:
        cur = self._conn.execute(
            "INSERT INTO conversations(mode, peer_key, room, created_at) "

        )

        row = self._conn.execute(
            "SELECT id, mode, peer_key, room, created_at "
            "FROM conversations WHERE id=?",
            (conversation_id,),

    def list_conversations(self) -> list[dict]:
        rows = self._conn.execute(
            "SELECT id, mode, peer_key, room, created_at "
            "FROM conversations ORDER BY id"
        ).fetchall()
        return [dict(r) for r in rows]

    def add_message(self, conversation_id: int, direction: str, body: bytes, status: str) -> int:
        cur = self._conn.execute(
            "INSERT INTO messages(conversation_id, direction, body, status, created_at) "
            "VALUES(?, ?, ?, ?, ?)",
            (conversation_id, direction, body, status, int(time.time())),
        )
        self._conn.commit()
        return cur.lastrowid

    def get_messages(self, conversation_id: int, limit: int | None = None, before_id: int | None = None) -> list[dict]:
        sql = (
            "SELECT id, conversation_id, direction, body, status, created_at "
            "FROM messages WHERE conversation_id=?"
        )
        args: list = [conversation_id]
        if before_id is not None:
            sql += " AND id < ?"
            args.append(before_id)
        sql += " ORDER BY id DESC"
        if limit is not None:
            sql += " LIMIT ?"
            args.append(limit)
        rows = self._conn.execute(sql, args).fetchall()
        return [dict(r) for r in reversed(rows)]
